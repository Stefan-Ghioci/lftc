main(){
    float v[100] = {10};
    float sum = 0.0;
    int n = 0;
    cout << "N = ";
    cin >> n;
    cout << "Dati numerele:";
    cout << endl;
    for(int i = 0; i < n; i = i+1){
        float x = 0.0;
        cin >> x;
        v[i] = x;
    }
    for(int i = 0; i < n; i = i+1){
        sum = sum + v[i];
    }
    cout << "Suma este ";
    cout << sum;
    cout << endl;
    return 0;
}
